\begin{eqnarray*}
  3^0 & = & 1 = 1 \\
  3^1 & = & 1 \times 3 = 3\\
  3^2 & = & 1 \times 3 \times 3 = 9 \\
  3^3 & = & 1 \times 3 \times 3 \times 3 = 27 \\
  \vdots & & \vdots \\
  3^n & = & 1 \times 3 \times 3 \times 3 \times \ldots \times 3 = 3^n \\
\end{eqnarray*}
